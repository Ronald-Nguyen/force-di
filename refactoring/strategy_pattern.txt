    private interface GreetingStrategy {
        String execute(String name);
    }

    private class AliceStrategy implements GreetingStrategy {
        public String execute(String name) {
            return 'Hello Alice';
        }
    }

    private class BobStrategy implements GreetingStrategy {
        public String execute(String name) {
            return 'Hello Bob';
        }
    }

    private static Map<String, GreetingStrategy> STRATEGIES = new Map<String, GreetingStrategy>{
        'Alice' => new AliceStrategy(),
        'Bob'   => new BobStrategy()
    };

    public static String greet(String name) {
        if (STRATEGIES.containsKey(name)) {
            return STRATEGIES.get(name).execute(name);
        }
        return 'Hello stranger';
    }
}
